# Format:
#
# Class Name: // Also actual name, unless specified. Spaces will show up in the name, but not class name.
#	Int: *Interfaces to implement*
#	Cat: *BagCategory*
#	BattleCat: *BattleBagCategory*, comma separate for multiple
#	Price: (leave out for -1, or unsellable)
#	ApplyEndTurn:
#		Function ending in "###" on a new line
# * // End this item
# // Any field not on the same line as its declarer (i.e. Apply) is considered a function

# // Not implemented: Casteliacone, Heal Powder, Lava Cookie

	Int: HoldItem, EndTurnEffect
	Cat: MISC
	Desc: An item to be held by a Pok\u00e9mon. It snaps the holder out of infatuation. It can be used only once.
	Price: 100

		boolean used = false;
		for (int i = 0; i < effects.length; i++)
		{
			Namesies s = effects[i];
			if (victim.hasEffect(s))
			{
				used = true;
				victim.getAttributes().removeEffect(s);
				b.addMessage(victim.getName() + " is no longer " + messages[i] + " due to its " + this.name + "!");
			}
		}
		if (used) victim.consumeItem(b);
		###
